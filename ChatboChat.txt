<script>
document.addEventListener('DOMContentLoaded', function() {
    // DOM Elements
    const chatInput = document.getElementById('chatInput');
    const sendButton = document.getElementById('sendButton');
    const chatMessages = document.getElementById('chatMessages');
    const chatHistory = document.getElementById('chatHistory');
    const typingIndicator = document.getElementById('typingIndicator');
    const clearChatBtn = document.getElementById('clearChatBtn');
    const currentMoodElement = document.getElementById('currentMood');
    
    // Current mood context
    let currentMood = null;
    // Track messages to prevent duplicates
    let sentMessages = new Set();
    
    // Load chat history
    loadChatHistory();
    
    // Send message on button click
    if (sendButton) {
        sendButton.addEventListener('click', sendMessage);
    }
    
    // Send message on Enter key (but allow Shift+Enter for new line)
    if (chatInput) {
        chatInput.addEventListener('keydown', function(e) {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                sendMessage();
            }
            
            // Auto-resize textarea
            this.style.height = 'auto';
            this.style.height = (this.scrollHeight) + 'px';
            this.style.height = Math.min(this.scrollHeight, 120) + 'px';
        });
        
        // Focus on input
        chatInput.focus();
    }
    
    // Clear chat history - FIXED
    if (clearChatBtn) {
        clearChatBtn.addEventListener('click', function() {
            if (confirm('Are you sure you want to clear all chat history? This cannot be undone.')) {
                clearChatHistory();
            }
        });
    }
    
    // Send message function - FIXED to prevent duplicates
    function sendMessage() {
        const message = chatInput.value.trim();
        
        if (!message) {
            alert('Please enter a message');
            return;
        }
        
        // Create unique ID for this message
        const messageId = Date.now() + '-' + Math.random().toString(36).substr(2, 9);
        
        // Check if already sent (prevent double clicks)
        if (sentMessages.has(messageId)) {
            return;
        }
        sentMessages.add(messageId);
        
        // Disable input and button
        chatInput.disabled = true;
        sendButton.disabled = true;
        
        // Clear input first
        const messageText = chatInput.value;
        chatInput.value = '';
        chatInput.style.height = 'auto';
        
        // Show typing indicator
        typingIndicator.style.display = 'flex';
        
        // Scroll to bottom
        chatMessages.scrollTop = chatMessages.scrollHeight;
        
        // Send to server
        fetch('{% url "chatbot:send_message" %}', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'X-CSRFToken': getCookie('csrftoken')
            },
            body: JSON.stringify({
                message: messageText,
                mood_context: currentMood
            })
        })
        .then(response => response.json())
        .then(data => {
            // Hide typing indicator
            typingIndicator.style.display = 'none';
            
            if (data.success) {
                // Update current mood
                if (data.mood_context) {

    // Helper functions
    function getCurrentTime() {
        const now = new Date();
        return now.toLocaleTimeString('en-GB', {
            timeZone: 'Asia/Kathmandu',
            hour: '2-digit',
            minute: '2-digit',
            hour12: false
        });
    }
    
    function formatTime(timeString) {
        if (!timeString) return getCurrentTime();
        
        // Try to parse the time string
        try {
            // If it's already in HH:MM format, return as is
            if (/^\d{1,2}:\d{2}$/.test(timeString)) {
                return timeString;
            }
            
            // Try parsing as ISO string
            const date = new Date(timeString);
            if (!isNaN(date.getTime())) {
                return date.toLocaleTimeString('en-GB', {
                    timeZone: 'Asia/Kathmandu',
                    hour: '2-digit',
                    minute: '2-digit',
                    hour12: false
                });
            }
            
            return timeString;
        } catch (e) {
            return timeString;
        }
    }

    // CSRF token helper
    function getCookie(name) {
        let cookieValue = null;
        if (document.cookie && document.cookie !== '') {
            const cookies = document.cookie.split(';');
            for (let i = 0; i < cookies.length; i++) {
                const cookie = cookies[i].trim();
                if (cookie.substring(0, name.length + 1) === (name + '=')) {
                    cookieValue = decodeURIComponent(cookie.substring(name.length + 1));
                    break;
                }
            }
        }
        return cookieValue;
    }
